## Assignment 2:
EQUI JOIN
Display JOB_ID, MANAGER_ID, SALARY, COMMISSION_PCT, DEPARTMENT_NAME for Sales Representatives.

Query :

SELECT E.JOB_ID,
       E.MANAGER_ID,
       E.SALARY,
       E.COMMISSION_PCT,
       D.DEPARTMENT_NAME
FROM EMPLOYEES E,
     DEPARTMENTS D
WHERE E.DEPARTMENT_ID=D.DEPARTMENT_ID
AND E.JOB_ID='SA_REP';

## Assignment 3:
Employees working in Seattle
The SCOTT assignment used DALLAS. HR uses cities stored in the LOCATIONS table.

Query :

SELECT E.FIRST_NAME,
       E.JOB_ID,
       E.DEPARTMENT_ID,
       D.DEPARTMENT_NAME
FROM EMPLOYEES E,
     DEPARTMENTS D,
     LOCATIONS L
WHERE E.DEPARTMENT_ID=D.DEPARTMENT_ID
AND D.LOCATION_ID=L.LOCATION_ID
AND L.CITY='Seattle';

## Assignment 5 :
OUTER JOIN
Include employees without managers.

Query :

SELECT E.FIRST_NAME AS Employee,
       E.EMPLOYEE_ID AS "Emp#",
       M.FIRST_NAME AS Manager,
       M.EMPLOYEE_ID AS "Mgr#"
FROM EMPLOYEES E
LEFT OUTER JOIN EMPLOYEES M
ON E.MANAGER_ID=M.EMPLOYEE_ID
ORDER BY E.EMPLOYEE_ID;

## Assignment 8 :
SELF JOIN
Employees hired before their manager.

Query :

SELECT E.FIRST_NAME,
       E.HIRE_DATE,
       M.FIRST_NAME,
       M.HIRE_DATE
FROM EMPLOYEES E
LEFT OUTER JOIN EMPLOYEES M
ON E.MANAGER_ID=M.EMPLOYEE_ID
WHERE M.HIRE_DATE>E.HIRE_DATE;

## Assignment 9 :
USING Clause
Display employees working as Stock Clerks.

Query :

SELECT EMPLOYEE_ID,
       FIRST_NAME,
       DEPARTMENT_NAME
FROM EMPLOYEES
JOIN DEPARTMENTS
USING(DEPARTMENT_ID)
WHERE JOB_ID='ST_CLERK';
